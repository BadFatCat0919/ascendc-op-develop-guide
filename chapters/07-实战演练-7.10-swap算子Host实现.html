4.6 swap 算子 Host 实现

Host 侧代码运行在 CPU 上，负责参数校验、Tiling 计算、内存管理和启动 Kernel。本节将逐步讲解 Host 侧的完整实现。

4.6.1 Host 侧完整代码

// swap_host.cpp
#include <iostream>
#include "acl/acl.h"
#include "swap_tiling_data.h"

// Kernel 入口函数声明（在 swap_kernel.cpp 中定义）
extern "C" void swap_kernel_entry(GM_ADDR x, GM_ADDR y, GM_ADDR tiling_ptr);

// Host 侧主函数
int swap_host(int N, float *x, float *y, int incx, int incy) {
    // ========== 1. 参数校验 ==========
    if (N <= 0) {
        std::cerr << "Error: N must be positive" << std::endl;
        return -1;
    }
    if (x == nullptr || y == nullptr) {
        std::cerr << "Error: x and y cannot be null" << std::endl;
        return -1;
    }
    if (incx <= 0 || incy <= 0) {
        std::cerr << "Error: incx and incy must be positive" << std::endl;
        return -1;
    }
    
    // ========== 2. Tiling 计算 ==========
    SwapTilingData tiling;
    tiling.totalN = N;
    
    int32_t maxCoreCount = 20;
    int32_t minElementsPerCore = 256;
    
    tiling.coreCount = (N + minElementsPerCore - 1) / minElementsPerCore;
    if (tiling.coreCount > maxCoreCount) {
        tiling.coreCount = maxCoreCount;
    }
    
    tiling.perCoreN = N / tiling.coreCount;
    tiling.remainder = N % tiling.coreCount;
    
    int32_t ubSizeBytes = 128 * 1024;  // 128KB
    tiling.tileSize = ubSizeBytes / sizeof(float) / 2;
    
    if (tiling.tileSize > tiling.perCoreN) {
        tiling.tileSize = tiling.perCoreN;
    }
    
    // ========== 3. 分配 Device 内存 ==========
    void *devX = nullptr;
    void *devY = nullptr;
    void *devTiling = nullptr;
    
    aclError ret;
    ret = aclrtMalloc(&devX, N * sizeof(float), ACL_MEM_MALLOC_HUGE_FIRST);
    if (ret != ACL_SUCCESS) {
        std::cerr << "Error: Failed to allocate device memory for X" << std::endl;
        return -1;
    }
    
    ret = aclrtMalloc(&devY, N * sizeof(float), ACL_MEM_MALLOC_HUGE_FIRST);
    if (ret != ACL_SUCCESS) {
        std::cerr << "Error: Failed to allocate device memory for Y" << std::endl;
        aclrtFree(devX);
        return -1;
    }
    
    ret = aclrtMalloc(&devTiling, sizeof(SwapTilingData), ACL_MEM_MALLOC_HUGE_FIRST);
    if (ret != ACL_SUCCESS) {
        std::cerr << "Error: Failed to allocate device memory for Tiling" << std::endl;
        aclrtFree(devX);
        aclrtFree(devY);
        return -1;
    }
    
    // ========== 4. 拷贝数据到 Device ==========
    ret = aclrtMemcpy(devX, N * sizeof(float), x, N * sizeof(float), 
                      ACL_MEMCPY_HOST_TO_DEVICE);
    ret = aclrtMemcpy(devY, N * sizeof(float), y, N * sizeof(float), 
                      ACL_MEMCPY_HOST_TO_DEVICE);
    ret = aclrtMemcpy(devTiling, sizeof(SwapTilingData), &tiling, 
                      sizeof(SwapTilingData), ACL_MEMCPY_HOST_TO_DEVICE);
    
    // ========== 5. 启动 Kernel ==========
    swap_kernel_entry<<<tiling.coreCount, 1>>>(devX, devY, devTiling);
    
    // ========== 6. 等待完成 ==========
    aclrtSynchronizeDevice();
    
    // ========== 7. 拷贝结果回 Host ==========
    ret = aclrtMemcpy(x, N * sizeof(float), devX, N * sizeof(float), 
                      ACL_MEMCPY_DEVICE_TO_HOST);
    ret = aclrtMemcpy(y, N * sizeof(float), devY, N * sizeof(float), 
                      ACL_MEMCPY_DEVICE_TO_HOST);
    
    // ========== 8. 释放 Device 内存 ==========
    aclrtFree(devX);
    aclrtFree(devY);
    aclrtFree(devTiling);
    
    return 0;
}

4.6.2 参数校验

参数校验是 Host 侧的第一步，确保输入参数合法。

// 1. N 必须为正数
if (N <= 0) {
    std::cerr << "Error: N must be positive" << std::endl;
    return -1;
}

// 2. 指针不能为空
if (x == nullptr || y == nullptr) {
    std::cerr << "Error: x and y cannot be null" << std::endl;
    return -1;
}

// 3. 步长必须为正数
if (incx <= 0 || incy <= 0) {
    std::cerr << "Error: incx and incy must be positive" << std::endl;
    return -1;
}

为什么要做参数校验？

  • 安全性：防止空指针导致段错误
  • 正确性：N ≤ 0 时没有数据需要处理
  • 调试：清晰的错误信息帮助定位问题

4.6.3 Tiling 计算

Tiling 计算是 Host 侧的核心逻辑，决定如何分配工作给多个 AI Core。

SwapTilingData tiling;
tiling.totalN = N;

// 1. 计算 Core 数量
int32_t maxCoreCount = 20;           // 硬件上限（910B 有 20 个 Core）
int32_t minElementsPerCore = 256;     // 每个 Core 至少处理 256 个元素
tiling.coreCount = (N + minElementsPerCore - 1) / minElementsPerCore;
if (tiling.coreCount > maxCoreCount) {
    tiling.coreCount = maxCoreCount;
}

// 2. 多核切分
tiling.perCoreN = N / tiling.coreCount;
tiling.remainder = N % tiling.coreCount;

// 3. UB 分块大小
int32_t ubSizeBytes = 128 * 1024;  // 128KB（每个 Buffer 的大小）
tiling.tileSize = ubSizeBytes / sizeof(float) / 2;  // 16384 个元素
if (tiling.tileSize > tiling.perCoreN) {
    tiling.tileSize = tiling.perCoreN;
}

4.6.4 内存分配与数据拷贝

Host 侧需要为 Kernel 分配 Device 内存，并把数据从 Host 拷贝到 Device。

内存分配

void *devX = nullptr;
void *devY = nullptr;
void *devTiling = nullptr;

// 分配 X 的 Device 内存
aclError ret = aclrtMalloc(&devX, N * sizeof(float), ACL_MEM_MALLOC_HUGE_FIRST);
if (ret != ACL_SUCCESS) {
    std::cerr << "Error: Failed to allocate device memory for X" << std::endl;
    return -1;
}

// 分配 Y 的 Device 内存
ret = aclrtMalloc(&devY, N * sizeof(float), ACL_MEM_MALLOC_HUGE_FIRST);
if (ret != ACL_SUCCESS) {
    aclrtFree(devX);  // 失败时释放已分配的内存
    return -1;
}

// 分配 TilingData 的 Device 内存
ret = aclrtMalloc(&devTiling, sizeof(SwapTilingData), ACL_MEM_MALLOC_HUGE_FIRST);
if (ret != ACL_SUCCESS) {
    aclrtFree(devX);
    aclrtFree(devY);
    return -1;
}

数据拷贝

// Host → Device（H2D）
aclrtMemcpy(devX, N * sizeof(float), x, N * sizeof(float), 
            ACL_MEMCPY_HOST_TO_DEVICE);
aclrtMemcpy(devY, N * sizeof(float), y, N * sizeof(float), 
            ACL_MEMCPY_HOST_TO_DEVICE);
aclrtMemcpy(devTiling, sizeof(SwapTilingData), &tiling, 
            sizeof(SwapTilingData), ACL_MEMCPY_HOST_TO_DEVICE);

4.6.5 Kernel 启动

Kernel 启动使用 <<<gridDim, blockDim>>> 语法，指定启动多少个 AI Core。

// 启动 Kernel：coreCount 个 AI Core，每个 Core 1 个线程块
swap_kernel_entry<<<tiling.coreCount, 1>>>(devX, devY, devTiling);

// 等待所有 Kernel 执行完成
aclrtSynchronizeDevice();

启动语法说明

参数含义示例
gridDim启动的 AI Core 数量tiling.coreCount（如 8）
blockDim每个 Core 的线程块数1（Ascend C 通常用 1）

4.6.6 结果回收

Kernel 执行完成后，需要把结果从 Device 拷贝回 Host，并释放 Device 内存。

// Device → Host（D2H）
aclrtMemcpy(x, N * sizeof(float), devX, N * sizeof(float), 
            ACL_MEMCPY_DEVICE_TO_HOST);
aclrtMemcpy(y, N * sizeof(float), devY, N * sizeof(float), 
            ACL_MEMCPY_DEVICE_TO_HOST);

// 释放 Device 内存
aclrtFree(devX);
aclrtFree(devY);
aclrtFree(devTiling);

4.6.8 关键 API 说明

API作用参数
aclrtMalloc分配 Device 内存(指针, 大小, 策略)
aclrtMemcpyHost↔Device 数据拷贝(dst, dstSize, src, srcSize, 方向)
aclrtFree释放 Device 内存(指针)
aclrtSynchronizeDevice等待所有 Kernel 完成无
<<<grid, block>>>启动 Kernel(Core数, 线程数)

Q&A：常见问题

Q1：为什么内存分配失败时要释放已分配的内存？

A：这是为了防止内存泄漏。如果 devX 分配成功但 devY 分配失败，不释放 devX 会导致这块内存永远无法被回收。正确的做法是"分配失败时，释放所有已分配的资源"。

Q2：aclrtMemcpy 的第二个和第四个参数有什么区别？

A：第二个参数是目标缓冲区的大小（dstSize），第四个参数是源数据的大小（srcSize）。两者通常相等，但 API 会检查 dstSize ≥ srcSize，防止越界写入。

Q3：<<>> 的含义是？

A：<<<gridDim, blockDim>>> 是启动 Kernel 的语法。gridDim = coreCount 表示启动 coreCount 个 AI Core，blockDim = 1 表示每个 Core 1 个线程块。在 Ascend C 中，每个 AI Core 独立执行 Kernel 代码，通过 GetBlockIdx() 获取自己的索引。
