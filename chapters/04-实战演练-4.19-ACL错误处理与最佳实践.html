4.15 ACL 错误处理与最佳实践

本节介绍 ACL 的错误处理机制、常见错误场景，以及在算子开发中积累的最佳实践经验。

错误处理

ACL API 的返回值类型是 aclError（即 int32_t），ACL_SUCCESS（0）表示成功，其他值表示各种错误。

aclGetRecentErrMsg

const char* aclGetRecentErrMsg();

作用：获取最近一次 ACL 错误的描述信息。当 API 返回非 0 值时，调用此函数可以获取人类可读的错误描述。

// 错误处理的标准模式
aclError ret = aclrtMalloc(&devPtr, size, ACL_MEM_MALLOC_HUGE_FIRST);
if (ret != ACL_SUCCESS) {
    printf("aclrtMalloc failed: %s\n", aclGetRecentErrMsg());
    // 处理错误...
    return ret;
}

推荐的错误检查宏

#define ACL_CHECK(call) do { \
    aclError ret = (call); \
    if (ret != ACL_SUCCESS) { \
        fprintf(stderr, "ACL Error at %s:%d - %s\n", \
                __FILE__, __LINE__, aclGetRecentErrMsg()); \
        exit(1); \
    } \
} while(0)

使用 ACL_CHECK 宏可以大幅简化错误处理代码：

// 使用宏前
aclError ret = aclrtSetDevice(0);
if (ret != ACL_SUCCESS) {
    fprintf(stderr, "aclrtSetDevice failed: %s\n", aclGetRecentErrMsg());
    return ret;
}

// 使用宏后
ACL_CHECK(aclrtSetDevice(0));

常用错误码

错误码含义常见原因
ACL_SUCCESS (0)成功—
ACL_ERROR_INVALID_PARAM参数无效传了空指针、非法尺寸等
ACL_ERROR_MEMORY_ALLOCATION内存分配失败HBM 空间不足
ACL_ERROR_INTERNAL内部错误驱动异常、设备故障
161xxx运行时配置错误环境变量缺失、版本不匹配
361xxxTiling 错误TilingData 计算有误
561xxxKernel 查找失败Kernel 名拼写错误或未编译

更详细的错误码列表请参考附录 C。

常见错误场景

场景 1：未初始化就调用 ACL API

// 错误：没有调用 aclInit
aclrtSetDevice(0);  // 返回错误码

// 正确
aclInit(nullptr);
aclrtSetDevice(0);

场景 2：Kernel 未同步就拷贝结果

// 错误：Kernel 是异步的，此时结果还没算完
my_kernel<<<coreNum, nullptr, stream>>>(devA, devB, devC, tilingGm);
aclrtMemcpy(hostC, size, devC, size, ACL_MEMCPY_DEVICE_TO_HOST);  // 数据不对！

// 正确：先同步再拷贝
my_kernel<<<coreNum, nullptr, stream>>>(devA, devB, devC, tilingGm);
aclrtSynchronizeStream(stream);  // 等待 Kernel 完成
aclrtMemcpy(hostC, size, devC, size, ACL_MEMCPY_DEVICE_TO_HOST);

场景 3：资源释放顺序错误

// 错误：先 Finalize 再释放内存（此时设备已经被重置）
aclFinalize();
aclrtFree(devPtr, size);  // 段错误！

// 正确：按依赖关系从内到外释放
aclrtFree(devPtr, size);
aclrtResetDevice(0);
aclFinalize();

场景 4：内存泄漏

// 错误：分配了但没有释放
aclrtMalloc(&devPtr, size, ACL_MEM_MALLOC_HUGE_FIRST);
// ... 程序结束，devPtr 没有释放

// 正确：成对使用
aclrtMalloc(&devPtr, size, ACL_MEM_MALLOC_HUGE_FIRST);
// ... 使用 devPtr ...
aclrtFree(devPtr, size);

最佳实践清单

资源管理

  1. 成对使用：aclInit/aclFinalize、aclrtSetDevice/aclrtResetDevice、aclrtMalloc/aclrtFree、aclrtCreateStream/aclrtDestroyStream 必须成对出现，否则会导致资源泄漏。
  2. 释放顺序：先释放 Stream/Event，再释放内存，最后 ResetDevice 和 Finalize。顺序反了可能导致段错误。
  3. 检查返回值：每个 ACL API 都可能失败，务必检查返回值。推荐使用 ACL_CHECK 宏。

内存管理

  1. 优先使用 Huge Page：ACL_MEM_MALLOC_HUGE_FIRST 性能更好，因为大页内存减少了 TLB miss。
  2. 减少拷贝次数：Host-Device 之间的数据拷贝是性能瓶颈，尽量减少不必要的拷贝。如果数据已经在 Device 上，就不要搬回 Host 再搬回去。
  3. 对齐要求：aclrtMalloc 分配的内存地址是 512B 对齐的，但 aclrtMemcpy 的拷贝长度没有对齐要求。

异步执行

  1. Kernel 是异步的：<<<>>> 启动 Kernel 后立即返回，不等 Kernel 执行完。必须在拷贝结果前调用 aclrtSynchronizeStream。
  2. Memcpy 是同步的：aclrtMemcpy 默认是同步的（阻塞等待拷贝完成）。如果需要异步拷贝，使用 aclrtMemcpyAsync。
  3. 多流并行：不同 Stream 上的 Kernel 可以并行执行。如果算子之间没有数据依赖，可以放在不同 Stream 上提升吞吐。

调试技巧

  1. 开启日志：设置环境变量 ASCEND_GLOBAL_LOG_LEVEL=1 可以开启 ACL 的详细日志输出。
  2. plog 日志：当 ACL API 返回错误时，查看 $HOME/ascend/log/ 目录下的 plog 日志，通常能找到更详细的错误原因。
  3. aclGetRecentErrMsg()：在错误发生后立即调用，获取人类可读的错误描述。

完整的 Host 侧模板

下面是一个使用直调方式的完整 Host 侧代码模板，把前面几节介绍的所有 API 串起来：

#include "acl/acl.h"
#include <cstdio>
#include <cstdlib>

#define ACL_CHECK(call) do { \
    aclError ret = (call); \
    if (ret != ACL_SUCCESS) { \
        fprintf(stderr, "ACL Error at %s:%d - %s\n", \
                __FILE__, __LINE__, aclGetRecentErrMsg()); \
        exit(1); \
    } \
} while(0)

int main() {
    const int n = 1024;
    const size_t size = n * sizeof(float);

    // 1. 初始化
    ACL_CHECK(aclInit(nullptr));
    ACL_CHECK(aclrtSetDevice(0));

    // 2. 准备 Host 数据
    float* hostA = (float*)malloc(size);
    float* hostB = (float*)malloc(size);
    float* hostC = (float*)malloc(size);
    for (int i = 0; i < n; i++) {
        hostA[i] = 1.0f;
        hostB[i] = 2.0f;
    }

    // 3. 分配 Device 内存
    void *devA = nullptr, *devB = nullptr, *devC = nullptr;
    ACL_CHECK(aclrtMalloc(&devA, size, ACL_MEM_MALLOC_HUGE_FIRST));
    ACL_CHECK(aclrtMalloc(&devB, size, ACL_MEM_MALLOC_HUGE_FIRST));
    ACL_CHECK(aclrtMalloc(&devC, size, ACL_MEM_MALLOC_HUGE_FIRST));

    // 4. 准备 Tiling 数据并传到 Device
    MyTilingData tilingData = { .totalLength = n, .tileSize = n / coreNum, ... };
    void* tilingGm = nullptr;
    ACL_CHECK(aclrtMalloc(&tilingGm, sizeof(MyTilingData), ACL_MEM_MALLOC_HUGE_FIRST));
    ACL_CHECK(aclrtMemcpy(tilingGm, sizeof(MyTilingData),
                           &tilingData, sizeof(MyTilingData),
                           ACL_MEMCPY_HOST_TO_DEVICE));

    // 5. 拷贝输入数据到 Device
    ACL_CHECK(aclrtMemcpy(devA, size, hostA, size, ACL_MEMCPY_HOST_TO_DEVICE));
    ACL_CHECK(aclrtMemcpy(devB, size, hostB, size, ACL_MEMCPY_HOST_TO_DEVICE));

    // 6. 创建 Stream 并启动 Kernel
    aclrtStream stream = nullptr;
    ACL_CHECK(aclrtCreateStream(&stream));
    my_kernel<<<coreNum, nullptr, stream>>>(devA, devB, devC, tilingGm);

    // 7. 等待 Kernel 完成
    ACL_CHECK(aclrtSynchronizeStream(stream));

    // 8. 拷贝结果回 Host
    ACL_CHECK(aclrtMemcpy(hostC, size, devC, size, ACL_MEMCPY_DEVICE_TO_HOST));

    // 9. 验证结果
    for (int i = 0; i < n; i++) {
        if (hostC[i] != 3.0f) {
            printf("Mismatch at %d: expected 3.0, got %f\n", i, hostC[i]);
            break;
        }
    }
    printf("All correct!\n");

    // 10. 清理资源
    ACL_CHECK(aclrtDestroyStream(stream));
    ACL_CHECK(aclrtFree(devA, size));
    ACL_CHECK(aclrtFree(devB, size));
    ACL_CHECK(aclrtFree(devC, size));
    ACL_CHECK(aclrtFree(tilingGm, sizeof(MyTilingData)));
    free(hostA); free(hostB); free(hostC);
    ACL_CHECK(aclrtResetDevice(0));
    ACL_CHECK(aclFinalize());

    return 0;
}

Q&A

Q：错误码 161xxx、361xxx、561xxx 分别代表什么？

A：这是昇腾运行时的错误码分段规则：161xxx 表示运行时配置错误（如环境变量缺失、版本不匹配），361xxx 表示 Tiling 错误（TilingData 计算有误），561xxx 表示 Kernel 查找失败（Kernel 名拼写错误或未编译）。详细的错误码说明参见附录 C。

Q：为什么资源释放要按特定顺序？

A：因为资源之间存在依赖关系。Stream 和 Event 依赖设备环境，Device 内存也依赖设备环境。如果先 aclrtResetDevice，设备环境被销毁，之后再释放 Stream 或内存就会访问已失效的资源，导致段错误。正确的顺序是：先释放 Stream/Event → 再释放内存 → 再 ResetDevice → 最后 Finalize。
