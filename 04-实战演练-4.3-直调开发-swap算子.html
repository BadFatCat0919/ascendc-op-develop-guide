4.3.1 swap 算子功能说明

swap 是 BLAS Level 1 标准中的算子，功能是交换两个向量的对应元素。执行 swap(x, y) 后，x 变成原来的 y，y 变成原来的 x。

数学定义

// BLAS 标准接口
void cblas_sswap(const int N, float *X, const int incX, float *Y, const int incY);

// 伪代码
for (int i = 0; i < N; i++) {
    float temp = X[i * incX];
    X[i * incX] = Y[i * incY];
    Y[i * incY] = temp;
}

参数说明

参数类型说明
Nint向量元素个数
Xfloat*第一个向量（输入/输出）
incXintX 的步长（通常为 1）
Yfloat*第二个向量（输入/输出）
incYintY 的步长（通常为 1）

示例

// 输入
X = [1.0, 2.0, 3.0, 4.0]
Y = [5.0, 6.0, 7.0, 8.0]

// 执行 swap(X, Y) 后
X = [5.0, 6.0, 7.0, 8.0]
Y = [1.0, 2.0, 3.0, 4.0]

4.3.3 TilingData 设计

TilingData 是 Host 和 Kernel 之间的"通信协议"——Host 计算好 Tiling 参数，通过 TilingData 传递给 Kernel。

swap_tiling_data.h 完整代码

// swap_tiling_data.h
#ifndef SWAP_TILING_DATA_H
#define SWAP_TILING_DATA_H

#include <cstdint>

// Tiling 数据结构定义
// Host 侧计算这些值，传递给 Kernel 侧使用
struct SwapTilingData {
    int32_t totalN;      // 总元素个数（N）
    int32_t perCoreN;    // 每个 AI Core 处理的元素个数
    int32_t remainder;   // 最后一个 Core 需要额外处理的元素个数
    int32_t tileSize;    // 每次搬运的元素个数（UB 分块大小）
    int32_t coreCount;   // 使用的 AI Core 数量
};

#endif // SWAP_TILING_DATA_H

Tiling 参数计算逻辑

假设 N = 10000，AI Core 数量 = 8，UB 容量 = 256KB

1. 多核切分：
   perCoreN = N / coreCount = 10000 / 8 = 1250
   remainder = N % coreCount = 10000 % 8 = 0

2. UB 分块：
   每个元素 4 字节（FP32）
   UB 可用空间 = 256KB / 2 = 128KB（双 Buffer 需要分两半）
   但 swap 用单 Buffer，所以 UB 可用空间 = 256KB
   tileSize = 128KB / 4 bytes / 2 buffers = 8192 个元素
   （这里 /2 是因为需要同时存 X 和 Y 两个向量）

3. 最终 TilingData：
   totalN = 10000
   perCoreN = 1250
   remainder = 0
   tileSize = 8192
   coreCount = 8

4.3.4 Kernel 实现

Kernel 是运行在 NPU AI Core 上的核函数，负责实际的数据搬运。

swap_kernel.cpp 完整代码

// swap_kernel.cpp
#include "kernel_operator.h"
#include "swap_tiling_data.h"

using namespace AscendC;

// Ascend C Kernel 类
class SwapKernel {
public:
    // 初始化函数：设置 Buffer 和 Tiling 参数
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, const SwapTilingData &tiling) {
        // 获取当前 Core 的索引
        int32_t coreIdx = GetBlockIdx();
        
        // 计算当前 Core 负责的数据范围
        int32_t offset = coreIdx * tiling.perCoreN;
        int32_t myN = tiling.perCoreN;
        
        // 最后一个 Core 需要处理余数
        if (coreIdx == tiling.coreCount - 1) {
            myN += tiling.remainder;
        }
        
        // 保存当前 Core 的参数
        this->myN = myN;
        this->tileSize = tiling.tileSize;
        
        // 设置 GM Buffer（指向全局内存）
        xGm.SetGlobalBuffer(reinterpret_cast<float*>(x + offset * sizeof(float)));
        yGm.SetGlobalBuffer(reinterpret_cast<float*>(y + offset * sizeof(float)));
        
        // 分配 UB Buffer（片上缓存）
        // 需要 4 个 Buffer：xIn, yIn（搬入）, xOut, yOut（搬出）
        // 但 swap 可以用交叉写回技巧，只需要 2 个 Buffer
        xBuf.Allocate(this->tileSize);
        yBuf.Allocate(this->tileSize);
    }
    
    // 主处理函数：循环处理所有数据块
    __aicore__ inline void Process() {
        int32_t processed = 0;
        
        // 循环处理，每次处理 tileSize 个元素
        while (processed < myN) {
            int32_t currentSize = (myN - processed < tileSize) 
                                  ? (myN - processed) 
                                  : tileSize;
            SingleIteration(processed, currentSize);
            processed += currentSize;
        }
    }
    
    // 单次迭代：处理一个数据块
    __aicore__ inline void SingleIteration(int32_t offset, int32_t size) {
        // 1. 从 GM 搬入到 UB（MTE2）
        DataCopy(xBuf, xGm[offset], size);
        DataCopy(yBuf, yGm[offset], size);
        
        // 2. 交叉写回：x 写回 y，y 写回 x（MTE3）
        // 这就是 swap 的核心：把 x 的数据写到 y 的位置，把 y 的数据写到 x 的位置
        DataCopy(xGm[offset], yBuf, size);  // y 的数据写到 x
        DataCopy(yGm[offset], xBuf, size);  // x 的数据写到 y
    }

private:
    GlobalTensor<float> xGm;    // X 向量的全局内存指针
    GlobalTensor<float> yGm;    // Y 向量的全局内存指针
    LocalTensor<float> xBuf;    // X 的片上缓存
    LocalTensor<float> yBuf;    // Y 的片上缓存
    int32_t myN;                // 当前 Core 需要处理的元素个数
    int32_t tileSize;           // 每次搬运的元素个数
};

// Kernel 入口函数（C 接口）
extern "C" __global__ __aicore__ void swap_kernel_entry(
    GM_ADDR x, 
    GM_ADDR y, 
    GM_ADDR tiling_ptr) 
{
    // 从 GM 读取 Tiling 数据
    SwapTilingData tiling;
    GM_ADDR tilingAddr = tiling_ptr;
    for (int i = 0; i < sizeof(SwapTilingData) / sizeof(int32_t); i++) {
        reinterpret_cast<int32_t*>(&tiling)[i] = 
            reinterpret_cast<int32_t*>(tilingAddr)[i];
    }
    
    // 创建 Kernel 实例并执行
    SwapKernel kernel;
    kernel.Init(x, y, tiling);
    kernel.Process();
}

关键 API 说明

API作用示例
GetBlockIdx()获取当前 AI Core 的索引int coreIdx = GetBlockIdx();
DataCopy(dst, src, count)数据搬运（GM↔UB）DataCopy(buf, gm[0], 256);
SetGlobalBuffer(ptr)设置全局内存指针gm.SetGlobalBuffer(ptr);
Allocate(size)分配片上 Bufferbuf.Allocate(256);

4.3.5 Host 侧实现

Host 侧代码运行在 CPU 上，负责参数校验、Tiling 计算和启动 Kernel。

swap_host.cpp 完整代码

// swap_host.cpp
#include <iostream>
#include "acl/acl.h"
#include "swap_tiling_data.h"

// Kernel 入口函数声明（在 swap_kernel.cpp 中定义）
extern "C" void swap_kernel_entry(GM_ADDR x, GM_ADDR y, GM_ADDR tiling_ptr);

// Host 侧主函数
int swap_host(int N, float *x, float *y, int incx, int incy) {
    // ========== 1. 参数校验 ==========
    if (N <= 0) {
        std::cerr << "Error: N must be positive" << std::endl;
        return -1;
    }
    if (x == nullptr || y == nullptr) {
        std::cerr << "Error: x and y cannot be null" << std::endl;
        return -1;
    }
    if (incx <= 0 || incy <= 0) {
        std::cerr << "Error: incx and incy must be positive" << std::endl;
        return -1;
    }
    
    // 目前只支持 incx = incy = 1 的情况
    if (incx != 1 || incy != 1) {
        std::cerr << "Warning: incx/incy != 1 not fully supported" << std::endl;
    }
    
    // ========== 2. Tiling 计算 ==========
    SwapTilingData tiling;
    tiling.totalN = N;
    
    // 获取可用的 AI Core 数量（910B 有 20 个 Core）
    int32_t maxCoreCount = 20;
    int32_t minElementsPerCore = 256;  // 每个 Core 至少处理 256 个元素
    
    // 计算实际使用的 Core 数量
    tiling.coreCount = (N + minElementsPerCore - 1) / minElementsPerCore;
    if (tiling.coreCount > maxCoreCount) {
        tiling.coreCount = maxCoreCount;
    }
    
    // 多核切分
    tiling.perCoreN = N / tiling.coreCount;
    tiling.remainder = N % tiling.coreCount;
    
    // UB 分块大小（假设 UB 可用 128KB，FP32 每个元素 4 字节）
    // 需要同时存 X 和 Y，所以 /2
    int32_t ubSizeBytes = 128 * 1024;  // 128KB
    tiling.tileSize = ubSizeBytes / sizeof(float) / 2;  // 16384 个元素
    
    // 确保 tileSize 不超过 perCoreN
    if (tiling.tileSize > tiling.perCoreN) {
        tiling.tileSize = tiling.perCoreN;
    }
    
    // ========== 3. 分配 Device 内存 ==========
    void *devX = nullptr;
    void *devY = nullptr;
    void *devTiling = nullptr;
    
    aclError ret;
    ret = aclrtMalloc(&devX, N * sizeof(float), ACL_MEM_MALLOC_HUGE_FIRST);
    if (ret != ACL_SUCCESS) {
        std::cerr << "Error: Failed to allocate device memory for X" << std::endl;
        return -1;
    }
    
    ret = aclrtMalloc(&devY, N * sizeof(float), ACL_MEM_MALLOC_HUGE_FIRST);
    if (ret != ACL_SUCCESS) {
        std::cerr << "Error: Failed to allocate device memory for Y" << std::endl;
        aclrtFree(devX);
        return -1;
    }
    
    ret = aclrtMalloc(&devTiling, sizeof(SwapTilingData), ACL_MEM_MALLOC_HUGE_FIRST);
    if (ret != ACL_SUCCESS) {
        std::cerr << "Error: Failed to allocate device memory for Tiling" << std::endl;
        aclrtFree(devX);
        aclrtFree(devY);
        return -1;
    }
    
    // ========== 4. 拷贝数据到 Device ==========
    ret = aclrtMemcpy(devX, N * sizeof(float), x, N * sizeof(float), 
                      ACL_MEMCPY_HOST_TO_DEVICE);
    ret = aclrtMemcpy(devY, N * sizeof(float), y, N * sizeof(float), 
                      ACL_MEMCPY_HOST_TO_DEVICE);
    ret = aclrtMemcpy(devTiling, sizeof(SwapTilingData), &tiling, 
                      sizeof(SwapTilingData), ACL_MEMCPY_HOST_TO_DEVICE);
    
    // ========== 5. 启动 Kernel ==========
    // <<>> 语法：gridDim = Core 数量，blockDim = 1（每个 Core 一个线程块）
    swap_kernel_entry<<>>(devX, devY, devTiling);
    
    // 等待 Kernel 执行完成
    aclrtSynchronizeDevice();
    
    // ========== 6. 拷贝结果回 Host ==========
    ret = aclrtMemcpy(x, N * sizeof(float), devX, N * sizeof(float), 
                      ACL_MEMCPY_DEVICE_TO_HOST);
    ret = aclrtMemcpy(y, N * sizeof(float), devY, N * sizeof(float), 
                      ACL_MEMCPY_DEVICE_TO_HOST);
    
    // ========== 7. 释放 Device 内存 ==========
    aclrtFree(devX);
    aclrtFree(devY);
    aclrtFree(devTiling);
    
    return 0;
}

关键 API 说明

API作用参数
aclrtMalloc分配 Device 内存(指针, 大小, 策略)
aclrtMemcpyHost↔Device 数据拷贝(dst, dstSize, src, srcSize, 方向)
aclrtFree释放 Device 内存(指针)
aclrtSynchronizeDevice等待所有 Kernel 完成无
<<>>启动 Kernel(Core数, 线程数)

4.3.6 交叉写回技巧

swap 的核心逻辑是"交换"，最直观的做法是用一个临时变量暂存，然后互相赋值。但在 NPU 上，我们可以用更高效的"交叉写回"技巧。

传统做法（需要额外 Buffer）

// 传统做法：需要 4 个 Buffer
LocalTensor<float> xBuf, yBuf, tempX, tempY;

// 1. 搬入
DataCopy(xBuf, xGm[offset], size);
DataCopy(yBuf, yGm[offset], size);

// 2. 交换（需要额外 Buffer）
DataCopy(tempX, xBuf, size);
DataCopy(tempY, yBuf, size);

// 3. 写回
DataCopy(xGm[offset], tempY, size);  // x = 原来的 y
DataCopy(yGm[offset], tempX, size);  // y = 原来的 x

交叉写回（只需要 2 个 Buffer）

// 交叉写回：只需要 2 个 Buffer
LocalTensor<float> xBuf, yBuf;

// 1. 搬入
DataCopy(xBuf, xGm[offset], size);  // xBuf = x 的数据
DataCopy(yBuf, yGm[offset], size);  // yBuf = y 的数据

// 2. 交叉写回（关键！）
DataCopy(xGm[offset], yBuf, size);  // 把 y 的数据写到 x 的位置
DataCopy(yGm[offset], xBuf, size);  // 把 x 的数据写到 y 的位置

为什么交叉写回更高效？

方案Buffer 数量UB 占用性能
传统做法4 个高（需要额外 temp Buffer）较慢
交叉写回2 个低（只需要 xBuf 和 yBuf）更快

交叉写回的核心思想是：不需要在 UB 内做交换，而是通过写回时的"交叉"来实现交换。xBuf 里存的是 x 的数据，但我们把它写到 y 的位置；yBuf 里存的是 y 的数据，但我们把它写到 x 的位置。这样就完成了交换，而且不需要额外的 Buffer。

  X (GM): [a, b, c, d, e, f, g, h]
  Y (GM): [1, 2, 3, 4, 5, 6, 7, 8]

4.3.8 关键设计决策

为什么用单 Buffer 而不是双 Buffer？

双 Buffer 的价值是让"搬运"和"计算"在不同 Buffer 上同时执行。但 swap 没有计算阶段——它只有搬运。搬运和搬运之间无法重叠（MTE2 和 MTE3 是独立的，但同一方向的搬运不能重叠），所以双 Buffer 对 swap 没有额外收益，反而浪费一倍 UB 空间。

为什么 Core 数量不是越多越好？

如果 N 很小（比如 N=100），用 20 个 Core 意味着每个 Core 只处理 5 个元素。启动 Core 的开销可能比实际计算还大。所以代码里设置了 minElementsPerCore = 256，确保每个 Core 至少处理 256 个元素。

为什么 tileSize 要除以 2？

因为需要同时存储 X 和 Y 两个向量的数据。UB 总容量是 256KB，但 xBuf 和 yBuf 各占一半，所以每个 Buffer 只能用 128KB。
